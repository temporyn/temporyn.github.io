# 디스크 크기 조정

LVM 환경에서 `/home` 볼륨을 제거해 확보한 공간을 `/`(root)로 확장하는 과정.

```bash
##################################################
# 파티션 확인
##################################################
$ df -h / /home
Filesystem             Size  Used Avail Use% Mounted on
/dev/mapper/rhel-root   39G  3.2G   35G   9% /
/dev/mapper/rhel-home   19G  165M   19G   1% /home

$ lsblk
NAME          MAJ:MIN RM  SIZE RO TYPE MOUNTPOINT
sda             8:0    0   64G  0 disk
├─sda1          8:1    0    1G  0 part /boot
└─sda2          8:2    0   63G  0 part
  ├─rhel-root 253:0    0   38G  0 lvm  /
  ├─rhel-swap 253:1    0  6.4G  0 lvm  [SWAP]
  └─rhel-home 253:2    0 18.6G  0 lvm

##################################################
# [root] 단일 사용자 모드 (다른 사용자 로그아웃)
##################################################
$ systemctl isolate rescue.target

##################################################
# 백업
##################################################
$ mkdir /home-backup
$ cp -aT /home /home-backup

##################################################
# 언마운트
##################################################
$ fuser -km /home
$ umount /home

##################################################
# fstab에서 /home 항목 주석 처리
##################################################
$ vi /etc/fstab
# /dev/mapper/rhel-home

##################################################
# LVM: home 제거 후 root로 확장
##################################################
$ lvremove /dev/mapper/rhel-home
$ lvextend -l +100%FREE /dev/mapper/rhel-root
$ xfs_growfs /

##################################################
# 백업 파일 복원
##################################################
$ mv /home-backup /home

##################################################
# 적용을 위해 재부팅
##################################################
$ reboot
```
