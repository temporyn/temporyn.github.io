# 프로그램 설치 가이드

Maven과 Java(SDKMAN!) 설치 방법.

## Maven

```bash
##################################################
# Version 3.6.3
##################################################
$ cd /tmp
$ wget https://archive.apache.org/dist/maven/maven-3/3.6.3/binaries/apache-maven-3.6.3-bin.tar.gz
$ tar -xzf apache-maven-3.6.3-bin.tar.gz
$ sudo mv apache-maven-3.6.3 /opt/utils/maven-3.6.3
$ echo 'export MAVEN_HOME=/opt/apache-maven-3.6.3' >> ~/.zshrc
$ echo 'export PATH=$MAVEN_HOME/bin:$PATH' >> ~/.zshrc
$ source ~/.zshrc
$ mvn -version
```

## Java (SDKMAN!)

```bash
##################################################
# SDKMAN! 설치
##################################################
$ curl -s "https://get.sdkman.io" | bash
$ source "$HOME/.sdkman/bin/sdkman-init.sh"

##################################################
# 내장 java 목록 사용 예
##################################################
$ sdk list java                     # 설치 가능한 버전 목록
$ sdk install java 17.0.13-tem      # Temurin 17 설치
$ sdk default java 17.0.13-tem      # 기본 버전 지정
$ sdk use java 17.0.13-tem          # 현재 셸에만 적용
$ java --version

##################################################
# 외부에서 받은 JDK 등록 (로컬 경로 지정)
##################################################
$ cd /tmp
$ wget https://repo.huaweicloud.com/java/jdk/8u202-b08/jdk-8u202-linux-x64.tar.gz
$ tar -xzf jdk-8u202-linux-x64.tar.gz
$ sdk install java 8.0.202-oracle /tmp/jdk1.8.0_202
$ sdk default java 8.0.202-oracle
```
