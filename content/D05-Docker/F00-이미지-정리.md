# 이미지 정리

```bash
# :latest 및 특정 레지스트리 이미지를 제외하고 삭제
docker images --format '{{.Repository}}:{{.Tag}}' \
  | grep -v ':latest$' \
  | grep -v '^registry:5000/base/' \
  | grep -v '^registry:5000/registry/' \
  | while read img; do
      # 사용 중(In Use) 이미지는 rmi 시 오류로 자동 스킵됨
      docker rmi "$img"
    done

# 빌드 캐시 정리
docker builder prune -a -f
```
